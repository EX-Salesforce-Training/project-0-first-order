public class CreditCardTriggers {
    public static void ExpiredCard (List<Credit_Card__c> listofCards){
       
     
        List<Task> addTask = new List<Task>(); //create task list
        for(Credit_Card__c c : listofCards){
            if (c.Expiration_Date__c < date.today()){ //test if date inputed is expired
                Task t = new Task(); //create task
                t.Subject = 'Notify Customer of Expired Card';
                t.Status = 'In Progress';
                t.WhatId = c.Id;
                t.OwnerId = c.OwnerId;
                t.ActivityDate = date.today() + 7;
                addTask.add(t);
        	}
    	}
        insert addTask;
    }
}
