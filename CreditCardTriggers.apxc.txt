public class CreditCardTriggers {
    public static void ExpiredCard (List<Credit_Card__c> listofCards){
        //if existing card is expired create a note/task/email
     
        List<Task> addTask = new List<Task>(); //create task list for bulk
        for(Credit_Card__c afterinsertCard : listofCards){
            if (afterinsertCard.Expiration_Date__c == date.today()){
                Task t = new Task();
                t.Subject = 'Notify Customer of Expired Card';
                t.Status = 'In Progress';     
        	}
    	}
        insert addTask;
    }
}
